Setup

This image will not work with a prefilled /pictures mount, lychee wants total control over this folder

Setup mysql/mariadb and account via the webui, accessible at http://SERVERIP:PORT More info at [lychee]({{ project_url }}).

# lychee - Generated by SH-Apps-Meta

version: "3.8"
services:
  lychee:
    image: linuxserver/lychee
    container_name: lychee
    restart: unless-stopped
    ports:
      #  http gui
      - 80:80
    volumes:
      #  Contains all relevant configuration files.
      - < /path/to/your/config>:/config
      #  Where lychee will store uploaded data.
      - < /path/to/your/pictures>:/pictures
    environment:
      #  User ID
      - PUID=100
      #  Group ID
      - PGID=100
      #  Specify a timezone to use EG Europe/London
      - TZ=Europe/London
      #  for specifying the database host
      - DB_HOST=mariadb
      #  for specifying the database port
      - DB_PORT=3306
      #  for specifying the database user
      - DB_USERNAME=lychee
      #  for specifying the database password
      - DB_PASSWORD=dbpassword
      #  for specifying the database to be used
      - DB_DATABASE=lychee
