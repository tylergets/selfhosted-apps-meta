# syncthing - Generated by SH-Apps-Meta

version: "3.8"
services:
  syncthing:
    image: linuxserver/syncthing
    container_name: syncthing
    restart: unless-stopped
    ports:
      #  Application WebUI
      - 8384:8384
      #  Listening port (TCP)
      - 22000/tcp:22000/tcp
      #  Listening port (UDP)
      - 22000/udp:22000/udp
      #  Protocol discovery
      - 21027/udp:21027/udp
    volumes:
      #  Configuration files.
      - < /path/to/your/config>:/config
      #  Data1
      - < /path/to/your/data1>:/data1
      #  Data2
      - < /path/to/your/data2>:/data2
    environment:
      #  User ID
      - PUID=100
      #  Group ID
      - PGID=100
      #  Specify a timezone to use EG Europe/London.
      - TZ=Europe/London
