Setup

Docker Compose
# medusa - Generated by SH-Apps-Meta

version: "3.8"
services:
  medusa:
    image: linuxserver/medusa
    container_name: medusa
    restart: unless-stopped
    ports:
      #  The port for the Medusa webui
      - 8081:8081
    volumes:
      #  Medusa config
      - < /path/to/your/config>:/config
      #  Download location
      - < /path/to/your/downloads>:/downloads
      #  TV Shows location
      - < /path/to/your/tv>:/tv
    environment:
      #  User ID
      - PUID=100
      #  Group ID
      - PGID=100
      #  Specify a timezone to use e.g. Europe/London
      - TZ=Europe/London
